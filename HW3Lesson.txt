    //*1. Написать программу, которая загадывает случайное число от 0 до 9, и пользователю дается 3 попытки угадать это число. При каждой попытке компьютер должен сообщить больше ли указанное пользователем число чем загаданное, или меньше. После победы или проигрыша выводится запрос – «Повторить игру еще раз? 1 – да / 0 – нет»(1 – повторить, 0 – нет).  // write your code here


    static Random random = new Random();
    static Scanner sc = new Scanner(System.in);

    static void main(String[] args) {
        guess_The_Number();}

        static void guess_The_Number () {
            do {
                int attempt = 0;
                int guess = -1;
                int number = random.nextInt(10);
                while (attempt < 3 && guess != number) {
                    System.out.print(
                            "Guess [" + (attempt + 1) + "] the number (0..9): ");
                    try {
                        guess = sc.nextInt();
                        if (number == guess) {
                            System.out.println("You won!");
                        } else {
                            System.out.println("Your number is " + ((guess > number) ?
                                    "greater" : "less."));
                            attempt++;
                        }
                    } catch (InputMismatchException ex) {
                        System.out.println("Input Mismatch Exception!");
                        sc.next();
                    }
                }
                if (attempt == 3)
                    System.out.println("You lost!");
                System.out.print("Repeat the game?\n[1 - yes / 0 - no]: ");
            } while (sc.next().equals("1"));

        }
    }
